①selsect语句基本查询
1.select   字段   from   表名;
2.别名:select   列名   as   值;
3.过滤:select   *   from   表名   where   字段;
②模糊查询
select   *   表名   where   字段   like   '%某';(以某结尾)；
                                                          '某%';(以某开头);
                                                          '%某%';(包含某);
                                                          '某_'; (下划线表示字符个数.);
③排序
select   *   from   表名   order   by   字段;(默认升序);
                                     order;   by;   DESC(降序);
④去重复
select   distinct   字段   from   表名;
⑤连接符
与Java类似;
⑥在...之间
between   值   and   值;/not   between   值   and   值;
⑦in与not int
⑧判断是否为空
select   *   from   表名   is   null   /   is   not   null;
⑨TOP(返回制定行数)
select   top   行数   返回字段   from   表名;
⑩聚合函数的过滤:having
select   列名   from   表名   group   by(分组)   having   过滤内容;
